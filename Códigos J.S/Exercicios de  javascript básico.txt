✅ 2. Verificador de idade

HTML

<input type="number" id="ano" placeholder="Ano de nascimento">
<button id="verificar">Verificar</button>
<p id="saida"></p>


JavaScript

document.querySelector('#verificar').addEventListener('click', function() {
    const ano = Number(document.querySelector('#ano').value);
    const atual = new Date().getFullYear();

    if (ano == 0 || ano > atual) {
        document.querySelector('#saida').innerHTML = "Ano inválido.";
    } else {
        const idade = atual - ano;
        document.querySelector('#saida').innerHTML = `Você tem ${idade} anos.`;
    }
});

✅ 5. Tabuada

HTML

<input type="number" id="num" placeholder="Digite um número">
<button id="gerar">Gerar tabuada</button>
<div id="resultado"></div>


JavaScript

document.querySelector('#gerar').addEventListener('click', function() {
    const numero = Number(document.querySelector('#num').value);
    const res = document.querySelector('#resultado');
    res.innerHTML = "";

    for (let i = 1; i <= 10; i++) {
        res.innerHTML += `${numero} x ${i} = ${numero * i}<br>`;
    }
});

✅ 6. Relógio digital

HTML

<h1 id="relogio"></h1>


JavaScript

function atualizarRelogio() {
    const agora = new Date();
    const horas = agora.getHours();
    const min = agora.getMinutes();
    const seg = agora.getSeconds();

    document.querySelector('#relogio').innerHTML =
        `${horas}:${min}:${seg}`;
}

setInterval(atualizarRelogio, 1000);
atualizarRelogio();
